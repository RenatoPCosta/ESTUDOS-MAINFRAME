/* PROGRAMA: FIZZBUZZ EM REXX */                      
/* CHATGPT(IA) */                                     
  MSGT = MSG('OFF')                                   
SAY "INFORME UM NUMERO:"                              
PULL N                                                
/* VERIFICA SE E NUMERICO */                          
IF DATATYPE(N, 'N') = 0 THEN DO                       
    SAY "VALOR INVALIDO. INFORME UM NUMERO INTEIRO."  
    EXIT                                              
END                                                   
DO I = 1 TO N                                         
    IF (I // 3 = 0) & (I // 5 = 0) THEN DO            
        SAY "FIZZBUZZ"                                
    END                                               
    ELSE IF I // 3 = 0 THEN DO                        
        SAY "FIZZ"                                    
    END                                               
    ELSE IF I // 5 = 0 THEN DO                        
        SAY "BUZZ"                                    
    END                                               
    ELSE DO                                           
        SAY I                                         
    END                                               
END
