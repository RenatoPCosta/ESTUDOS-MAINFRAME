  SAY QUEUED()

    DO X=1 TO QUEUED()

      PULL VARLINHAA                        
      SAY VARLINHAA                         
    END

/* PUSH -> FIFO(FIRST IN FIRST OUT) */      
NEWSTACK                                    
  SAY QUEUED()                              
  PUSH 'P1'                                 
  PUSH 'P2'                                 
 DO J=1 TO QUEUED()                         
  SAY QUEUED()                              
  PULL VARLINHA ; SAY VARLINHA              
 END                                        
DELSTACK
